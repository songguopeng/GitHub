#include<iostream>
#include<list>
#include<algorithm>
#include<iterator>
using  namespace std;

int main()
{
	list<int>l;
	int a;
	while (cin >> a && a != 0)
		l.push_back(a);
	l.sort();//list不能用sort排序，sort排序需要的随机迭代器，方便排序算法中的数据交换，list是双向迭代器，想要排序必须用list自身提供的sort排序
	copy(l.begin(), l.end(), ostream_iterator<int>(cout, "\t"));
	cout << endl;

	system("pause");



	return 0;
}
