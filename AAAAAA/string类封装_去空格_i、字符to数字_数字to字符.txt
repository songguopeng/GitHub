#include<iostream>
#include<sstream>
#include<string>

using namespace std;

class ext_string :public string
{
public:
	int GetInt(string str)
	{
		stringstream os;
		os << str;
		int n;
		os >> n;
		return n;
	}
	string GetString(int n)
	{
		stringstream os;
		string str;
		os << n;
		os >> str;
		return str;
	}
	string GetTrim(string strText)
	{
		strText.erase(0, strText.find_first_not_of(" "));
		strText.erase(strText.find_last_not_of(" ") + 1);
		return strText;
	}
};

int main(int argc,char*argv[])
{
	ext_string str;
	int n = str.GetInt("123424");
	string s = str.GetString(123);
	string s1 = str.GetTrim("   4324   ");
	cout << n << endl;
	cout << s << endl;
	cout << s1 << endl;
	getchar();
	return 0;
}
