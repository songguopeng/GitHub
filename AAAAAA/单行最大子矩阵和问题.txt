#include<iostream>
#include<vector>
#include<functional>
#include<iterator>
#include<algorithm>
using namespace std;

int main()
{
	int n;
	while (cin >> n)
	{
		int max_sum;
		int *A = new int[n];
		
		for (int i = 0; i < n; i++)
			cin >> A[i];
		int *tmp = new int[n];
		max_sum=tmp[0] = A[0];
		for (int i = 1; i <n; i++)
		{
			tmp[i] = max(A[i], A[i] + tmp[i - 1]);
			if (max_sum < tmp[i])
				max_sum = tmp[i];
		}
		cout << max_sum << endl;
		delete[] A;
		delete[] tmp;
	}
	return 0;
}
