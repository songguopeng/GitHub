#include<iostream>
#include<algorithm>
#include<iostream>
#include<iterator>
#include<functional>
#include<vector>

using namespace std;
//not1 是针对一元函数而言，not2是针对二元函数而言，
int main()
{
	vector<int> vt;
	int a;
	while (cin >> a&&a)
	{
		vt.push_back(a);
	}
	int nCount = count_if(vt.begin(), vt.end(), bind2nd(less<int>(), 4));
	cout << "小于4个数:" << nCount<<endl;
	nCount = count_if(vt.begin(), vt.end(), bind1st(less<int>(), 4));
	cout << "大于4个数:" << nCount << endl;
	nCount = count_if(vt.begin(), vt.end(), not1(bind2nd(less<int>(), 4)));
	cout<< "大于等于4个数:" << nCount << endl;
	nCount = count_if(vt.begin(), vt.end(), not1(bind1st(less<int>(), 4)));
	cout << "小于等于4个数:" << nCount << endl;
	sort(vt.begin(), vt.end(), not2(less_equal<int>()));
	copy(vt.begin(), vt.end(), ostream_iterator<int>(cout, "\t"));
	cout << endl;

	system("pause");
	return 0;

	

}
